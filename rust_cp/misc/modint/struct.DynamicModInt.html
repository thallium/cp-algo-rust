Struct rust_cp::misc::modint::DynamicModInt

source · [−]
#[repr(transparent)]
pub struct DynamicModInt<I> { /* private fields */ }
Expand description

Represents ℤ/mℤ where m is a dynamic value.

Corresponds to atcoder::dynamic_modint in the original ACL.

Example

ⓘ
use rust_cp::misc::modint::ModInt as Mint;
use proconio::{input, source::once::OnceSource};

input! {
    from OnceSource::from("3 3 7\n"),
    a: u32,
    b: u32,
    m: u32,
}

Mint::set_modulus(m);
let a = Mint::new(a);
let b = Mint::new(b);

println!("{}", a * b); // `2`
